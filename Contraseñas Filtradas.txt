ACCION OrdenarContraseñasPorFrecuencia ES
    AMBIENTE
        // Vectores paralelos, una misma posicion representa que los valores en ambos arreglos estan relacionados
        contrasenias: ARREGLO[0..n-1] de ALFANUMERICO
        veces: ARREGLO[0..n-1] de ENTERO // (cantidad de veces encontrada)
        
        n, j, aux_veces: ENTERO      // n ← longitud de los arreglos
        aux_pwd: ALFANUMERICO
        Badera: Booleana
    PROCESO
        // Consideramos que nuestros arreglos fueron previamente cargados en un proceso anterior
        // Ordenamiento por intercambio (descendente)
        Badera:= Falso
        Mientras NO Bandera Hacer
            Bandera:= Verdadero
            PARA j:= 1 HASTA n-1 HACER
                // Si la contraseña en j aparece más veces que la de i, intercambiamos
                SI veces[j] < veces[j+1] ENTONCES
                    // Intercambiar en veces
                    aux_veces := veces[j]
                    veces[j]  := veces[j+1]
                    veces[j+1]  := aux_veces

                    // Intercambiar la contraseña correspondiente
                    aux_pwd := contrasenias[j]
                    contrasenias[j] := contrasenias[j+1]
                    contrasenias[j+1] := aux_pwd

                    Bandera := Falso
                FIN_SI
            FIN_PARA
        FIN_MIENTRAS
        // Al final, contrasenias[] y veces[] están ordenadas por veces[] (descendente)
FIN_ACCION

ACCION OrdenarContraseniasPorFrecuencia
  // Precondici칩n: contrasenias y veces tienen la misma longitud n >= 0
  contrasenias: ARREGLO[0..n-1] de ALFANUMERICO
  veces:        ARREGLO[0..n-1] de ENTERO
  n, i, j: ENTERO
  aux_veces: ENTERO
  aux_pwd: ALFANUMERICO

  //n = longitud(veces)  // asumir igual a longitud(contrasenias)
  PARA i := 0 HASTA n-2 HACER
    PARA j := i+1 HASTA n-1 HACER
      // Si existe una contrase침a con m치s apariciones, la ponemos en i
      SI veces[i] < veces[j] ENTONCES
        // intercambiar en veces
        aux_veces := veces[i]
        veces[i] := veces[j]
        veces[j] := aux_veces
        // intercambiar la contrase침a correspondiente
        aux_pwd := contrasenias[i]
        contrasenias[i] := contrasenias[j]
        contrasenias[j] := aux_pwd
      FIN_SI
    FIN_PARA
  FIN_PARA
FIN_ACCION
